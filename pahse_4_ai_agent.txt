🔁 MAIN GRAPH CONTROLLER

This is your LangGraph-like engine

def run_agent(query):

    memory = get_memory(query)

    # 1. PLAN
    plan = planner_node(query, memory)

    final_context = ""

    # 2. EXECUTE LOOP
    for step in plan:
        result = executor_node(step, final_context)

        tool_result = tool_node(step)
        if tool_result:
            result += "\n" + str(tool_result)

        final_context += "\n" + result

    # 3. FINAL ANSWER
    answer = generate_response(
        "llama3:8b-instruct-q4_0",
        f"""
        FINAL QUESTION: {query}
        CONTEXT: {final_context}
        """
    )

    # 4. CRITIC LOOP
    score = critic_node(query, answer)

    if score < 7:
        print("🔁 Refining answer...")

        improved = generate_response(
            "llama3:8b-instruct-q4_0",
            f"Improve this answer:\n{answer}"
        )

        answer = improved
        score = critic_node(query, answer)

    # 5. MEMORY
    memory_node(query, plan, answer, score)

    return answer
